class Solution:
    def minSubArrayLen(self, k: int, nums: List[int]) -> int:
        left=0
        n=len(nums)
        s=0
        ans=float('inf')
        for right in range(n):
            s+=nums[right]
            while s>=k:
                ans=min(ans,right-left+1)
                s-=nums[left]
                left+=1
        return ans if ans!=float('inf') else 0
