{
    //Function to find the maximum money the thief can get.
    public int FindMaxSum(int arr[], int n)
    {
        // Your code here
        int[] dp=new int[n+1];
        dp[0]=0;
        for(int i=1;i<=n;i++){
            dp[i]=arr[i-1];
        }
        for(int i=2;i<=n;i++)
        {
            dp[i]=Math.max(dp[i-1],dp[i]+dp[i-2]);
        }
        return dp[n];
        
    }
}
