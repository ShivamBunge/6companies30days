from collections import defaultdict
class Solution:
    #use mp to keep track of repeating elements and             
    #q to check char in sequence
	def FirstNonRepeating(self, A):
		# Code here
		ans=""
		q=[]
		mp=defaultdict(int)
		for i in range(len(A)):
		    c=A[i]
		    mp[c]+=1
    		q.append(c)
    		while q:
    		    if q[0] in mp and mp[q[0]]>1:
    		        q.pop(0)
    		    else:
    		        break
    		if not q:
    		    ans+="#"
    		else:
    		    ans+=q[0]
        return ans
