
class Solution:
    def isValid(self, board):
        # code here
        N = 9

        # Use hash set to record the status
        rows = [set() for _ in range(N)]
        cols = [set() for _ in range(N)]
        boxes = [set() for _ in range(N)]

        for r in range(N):
            for c in range(N):
                val = board[r][c]
                # Check if the position is filled with number
                if val == 0:
                    continue

                # Check the row
                if val in rows[r]:
                    return 0
                rows[r].add(val)

                # Check the column
                if val in cols[c]:
                    return 0
                cols[c].add(val)

                # Check the box
                idx = (r // 3) * 3 + c // 3
                if val in boxes[idx]:
                    return 0
                boxes[idx].add(val)

        return 1
