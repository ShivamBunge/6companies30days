    public boolean winnerOfGame(String colors) {
        int a = 0, b = 0;
        for (int i = 2; i < colors.length(); ++i) {
            if (colors.charAt(i - 2) == colors.charAt(i - 1) && colors.charAt(i - 1) == colors.charAt(i)) {
                if (colors.charAt(i) == 'A') {
                    ++a;
                }else {
                    ++b;
                }
            }
        }
        return a > b;
    }
